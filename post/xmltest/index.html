XML解析——SAX方式

08 Aug 2019, 10:22

XML样例

<?xml version="1.0" encoding="UTF-8" ?>
<persons>
    <person>
        <name>至尊宝</name>
        <age>9000</age>
    </person>
    <person>
        <name>白晶晶</name>
        <age>7000</age>
    </person>
</persons>

SAX解析方式

注意： 将读到的标签qName保存在临时变量中，用来帮助存储数据contents。还需在适当的时候清空此临时变量。 进行存入List的操作判断依据在于判断qName！！！不是tag！！！

public class XMLTest {
    public static void main(String[] args) throws ParserConfigurationException, SAXException, IOException {
        //1、获取解析工厂
        SAXParserFactory factory=SAXParserFactory.newInstance();
        
        //2、从解析工厂获取解析器
        SAXParser parse =factory.newSAXParser();
        
        //3、编写处理器
        //4、加载文档 Document 注册处理器
        PersonHandler handler=new PersonHandler();
        
        //5、解析
        parse.parse(XMLPATH,handler);// XMLPATH为XML文件路径

        //6、获取数据
        List<Person> list = handler.getList();
        for(Person ele:list){
            System.out.println(ele.getName()+" "+ele.getAge());
        }
    }
}

class PersonHandler extends DefaultHandler {
    private List<Person> list;// 容器
    private Person person;
    private String tag;// 存储标签

    public List<Person> getList() {
        return list;
    }

    @Override
    public void startDocument() throws SAXException {
        // 当开始解析文档时，初始化容器
        list = new ArrayList<>();
    }

    @Override
    public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
        // 当解析到person标签的时候，对person初始化
        if (qName!=null){
            tag = qName;
            if (tag.equals("person")){
                person = new Person();
            }
        }
    }

    @Override
    public void characters(char[] ch, int start, int length) throws SAXException {
        String contents = new String(ch,start,length).trim();
        // 根据读到的标签，按需存储到Person对象中
        if (tag!=null) {
            if (tag.equals("name")) {
                person.setName(contents);
            } else if (tag.equals("age")) {
                person.setAge(Integer.valueOf(contents));
            }
        }
    }

    @Override
    public void endElement(String uri, String localName, String qName) throws SAXException {
        // 当解析到person标签时，将person加入到容器中
        // 请注意此时的判断依据为qName不是tag！！！
        if (qName!=null) {
            if (qName.equals("person")) {
                list.add(person);
            }
        }
        tag = null;// 将tag丢弃，避免读到空格将contents覆盖
    }

    @Override
    public void endDocument() throws SAXException {
    }
}

class Person {
    private String name;
    private int age;

    public Person() {
    }

    public Person(String name, int age) {
        this.name = name;
        this.age = age;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }
}
