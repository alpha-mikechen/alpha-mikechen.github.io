1. 栈的实现 (数组/链表)

数组实现栈

class ArrayStack {
    private int maxSize;
    private int [] stack;
    private int top = -1;//栈顶指针初始值为-1

    public ArrayStack(int maxSize) {
        this.maxSize = maxSize;
        stack = new int [this.maxSize];
    }

    public boolean isFull(){
        return top == maxSize-1;
    }

    public boolean isEmpty(){
        return top == -1;
    }

    public void push(int num){
        if (isFull()){
            System.out.println("栈已满");
            return;
        }
        stack[++top] = num;
    }

    public int pop(){
        if (isEmpty()){
            throw new RuntimeException("栈为空");
        }
        int value = stack[top--];
        return value;
    }

    public void showAll(){
        if (isEmpty()){
            throw new RuntimeException("栈为空");
        }
        for(int i=top;i>=0;i--){
            System.out.printf("stack[%d]=%d\n",i,stack[i]);
        }
    }
}

链表实现栈

class LinkedListStack {
    StackNode head = null;
    StackNode top = null;

    public void push(StackNode newNode){
        // 将第一个新节点作为头节点
        if (head == null){
            head = newNode;
            top = head;
        }
        // 将其余新节点链在最前面
        else {
            newNode.next = top;
            top = newNode;
        }
    }

    public int pop(){
        if (top == null){
            throw new RuntimeException("栈空");
        }
        int value = top.num;
        top = top.next;
        return value;
    }

    public void showAll(){
        StackNode temp = top;
        while (temp != null){
            System.out.println(temp.num);
            temp = temp.next;
        }
    }

}

2. 栈的应用 - 字符串计算器

实现理论

  1. 对字符串顺序扫描，数字字符存入数栈，运算符存入符号栈
  2. 扫描到运算符时，若符号栈为空，则直接入栈;若符号栈不为空，则判断符号优先级
  3. 若当前符号优先级小于等于栈中运算符，从数栈中pop出2个数，从符号栈中pop出一个符号，做运算。运算结果push进数栈，当前符号push进符号栈
  4. 若当前符号优先级大于栈中运算符，直接push进符号栈
  5. 当字符串被扫描完毕，顺序pop出两个栈中的数据，做运算
  6. 当数栈中仅剩最后一个数字时，其为最终运算结果

注意： 核心实现在于数栈，符号栈。运算符优先级。运算方式。

public int peak(){
    return stack[top];

// 判断是否是运算符
public boolean isOper(int ch){
    return ch == '+' || val == '-' || val == '*' || val == '/';
}

// 判断运算符优先级（自己定义）
public int priority(int oper){
    if (oper == '*' || oper == '/'){
        return 1;
    }
    else if (oper == '+' || oper == '-'){
        return 0;
    } else {
        return -1;
    }
}

// 计算
public int calculate(int num1, int num2, int oper){
    int result = 0;
    switch (oper){
        case '+': {
            result = num2 + num1;
            break;
        }
        case '-': {
            result = num2 - num1;
            break;
        }
        case '*': {
            result = num2 * num1;
            break;
        }
        case '/': {
            result = num2 / num1;
            break;
        }
    }
    return result;
}

计算器的实现

public class CalculatorTest {
    public static void main(String[] args) {
        String expression = "3+5*8/2-2";// 表达式
        Stack numStack = new Stack(expression.length());// 数栈
        Stack operStack = new Stack(expression.length());// 符号栈
        int num1;
        int num2;
        int oper;// 运算符
        int res; // 运算结果
        int index = 0;// 索引
        char ch;// 字符串的各个元素
        while (true){
            //依次得到字符串的每一个元素
            ch = expression.substring(index,++index).charAt(0);

            //若元素为运算符
            if (operStack.isOper(ch)){
                //若符号栈为空，直接入栈
                if (operStack.isEmpty()){
                    operStack.push(ch);
                }
                // 若符号栈不为空
                // 如果当前符号优先级小于等于符号栈的符号
                else if (operStack.priority(ch) <= operStack.priority(operStack.peak())){
                    num1 = numStack.pop();
                    num2 = numStack.pop();
                    oper = operStack.pop();
                    res = numStack.calculate(num1,num2,oper);
                    numStack.push(res);// 把运算结果入数栈
                    operStack.push(ch);// 把当前符号入符号栈
                }
                // 如果当前符号优先级大于符号栈的符号
                else if (operStack.priority(ch) > operStack.priority(operStack.peak())){
                    operStack.push(ch);
                }
            }
            //若元素为数字
            else {
                //直接入数栈，注意字符转数字
                numStack.push(ch-48);
            }
            // 判断index是否索引到字符串末尾
            if (index >= expression.length()){
                break;
            }
        }
        while (true){
            if (operStack.isEmpty()){
                break;
            }
            num1 = numStack.pop();
            num2 = numStack.pop();
            oper = operStack.pop();
            res = numStack.calculate(num1,num2,oper);
            numStack.push(res);
        }
        System.out.println("表达式的结果是: "+numStack.peak());
    }
}
