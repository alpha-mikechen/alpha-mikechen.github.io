Java数据结构与算法: LinkedList

16 Jul 2019, 11:07

1. 单向链表

概念

  • n个节点链结成的一个链表，即为线性表的链式存储结构。因为此链表中的每个节点中只包含一个数据域和一个指针域，所以叫做单链表。
  • 链表中第一个节点的存储位置叫做头指针
  • 任意一个节点的指针域中存储了下一个节点的地址

实现理论(关键部分)

  • 头节点是不能被修改的，因此遍历链表时需要用到临时变量temp
  • 由于是单向链表，执行增删操作时，需将temp定位到待操作位置的前一个；在循环遍历中通过标识flag来确定是否找到操作位置

编程实现


class SinglyLinkedList{
    //初始化一个头节点，不存放数据；头节点不能动
    private Node headNode =new Node(0,"","");

    //始终在链表尾部插入新节点
    public void addLast(Node newNode){
        //首先遍历链表找到链表尾部
        Node temp = headNode;
        while (true){
            if (temp.next == null){
                break;
            }
            temp = temp.next;
        }
        //while循环结束时，temp就指向了链表的最后
        temp.next = newNode;
    }
    
    //根据节点中的编号顺序，在正确的位置插入新节点
    public void addByOrder(Node newNode){
        Node temp = headNode;
        boolean flag = false;
        while (true){
            if (temp.next == null){
                break;
            }
            if (temp.next.num > newNode.num){
                break;
            } else if (temp.next.num == newNode.num){
                flag = true;
                break;
            }
            temp = temp.next;
        }
        if (flag){
            System.out.println("编号已存在");
        } else {
            newNode.num = temp.next;
            temp.next = newNode;
        }
    }
    
    //根据新节点中的编号修改节点信息，编号不允许被修改
    public void update(Node newNode){
        if (headNode.next == null){
            System.out.println("链表为空");
            return;
        }
        Node temp = headNode.next;
        boolean flag = false;
        while (true){
            if (temp == null){//检查是否以及遍历到链表尾
                flag = false;
                System.out.println("链表中没有该编号");
                break;
            }
            if (temp.num == newNode.num){
                flag = true;
                break;
            }
            temp = temp.next;
        }
        if (flag){
            temp.name = newNode.name;
            temp.nickName = newNode.nickName;
        }
    }
    
    //根据编号移除节点
    public void remove(int num){
        if (headNode.next == null){
            System.out.println("链表为空");
            return;
        }
        Node temp = headNode;
        boolean flag = false;
        while (true){
            if (temp == null){
                System.out.println("链表中没有该编号的节点");
                break;
            }
            if (temp.next.num == num){
                flag = true;
                break;
            }
            temp = temp.next;
        }
        if (flag){
            temp.next = temp.next.next;
        }
    }

    //显示链表
    public void show(){
        //首先判断头节点的指针域是否为空
        if (headNode.next == null){
            System.out.println("链表为空");
            return;
        }
        else {
            //若不为空，则循环遍历之后的节点
            Node temp = headNode.next;
            while (true){
                if (temp == null){
                    break;
                }
                else {
                    System.out.println(temp.toString());
                    temp = temp.next;
                }
            }
        }
    }
}

class Node{
    int num;
    String  name;
    String  nickName;
    Node next;

    public Node(int num, String name, String nickName) {
        this.num = num;
        this.name = name;
        this.nickName = nickName;
    }

    @Override
    public String toString() {
        return "Node{" +
                "num=" + num +
                ", name='" + name + '\'' +
                ", nickName='" + nickName + '\'' +
                '}';
    }
}
