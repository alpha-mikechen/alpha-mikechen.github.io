class SinglyLinkedList{
    //初始化一个头节点，不存放数据；头节点不能动
    private Node headNode =new Node(0,"","");

    //在链表尾部插入新节点
    public void add(Node newNode){
        //首先遍历链表找到链表尾部
        Node temp = headNode;
        while (true){
            if (temp.next == null){
                break;
            }
            else {
                temp = temp.next;
            }
        }
        //while循环结束时，temp就指向了链表的最后
        temp.next = newNode;
    }

    //显示链表
    public void show(){
        //首先判断头节点的指针域是否为空
        if (headNode.next == null){
            System.out.println("链表为空");
            return;
        }
        else {
            //若不为空，则循环遍历之后的节点
            Node temp = headNode.next;
            while (true){
                if (temp == null){
                    break;
                }
                else {
                    System.out.println(temp.toString());
                    temp = temp.next;
                }
            }
        }
    }
}

class Node{
    int num;
    String  name;
    String  nickName;
    Node next;

    public Node(int num, String name, String nickName) {
        this.num = num;
        this.name = name;
        this.nickName = nickName;
    }

    @Override
    public String toString() {
        return "Node{" +
                "num=" + num +
                ", name='" + name + '\'' +
                ", nickName='" + nickName + '\'' +
                '}';
    }
}
